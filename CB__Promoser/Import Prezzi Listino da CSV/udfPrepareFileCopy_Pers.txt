/* 
*
*	Ritorna il comando CMD da lanciare sul pc del cliente per copiare il file selezionato sul server (cartella condivisa)
*
*
*/
CREATE FUNCTION [udfPrepareFileCopy_Pers]
(
	@sFilePath varchar(255)
	, @iActualTimestamp bigint
)
RETURNS varchar(MAX)
AS
BEGIN
	Declare @sCmd varchar(MAX)
	Declare @sFinalFilePath varchar(255) = '\\SRV-PROMOSER\BEEP\CUSTOM\tmpExtraction\' + Cast(@iActualTimestamp as varchar(20)) + '.csv'
	Set @sFilePath = IsNull(@sFilePath, '')
	
	If Len(LTrim(RTrim(@sFilePath))) > 0
	Begin
		Set @sCmd = Concat(
			'"', @sFilePath, '" '
			, '"', @sFinalFilePath, '" '
		)
		
	End
	
	Return @sCmd
END

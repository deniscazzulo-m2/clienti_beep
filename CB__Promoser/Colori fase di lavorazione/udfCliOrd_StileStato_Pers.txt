CREATE FUNCTION [dbo].[udfCliOrd_StileStato_Pers]
(
	@sStato varchar(32)
)
RETURNS varchar(1024)
AS
BEGIN
Declare @sStile varchar(64)

If UPPER(@sStato) like 'PREPBOZ%'
	Set @sStile = 'green, (white)'
Else If UPPER(@sStato) like 'BOZINV%'
	Set @sStile = 'red'
Else If UPPER(@sStato) like 'BOZCONF%' 
	Set @sStile = 'purple, (white)'
Else If UPPER(@sStato) like 'LAV%' 
	Set @sStile = 'yellow'
Else If UPPER(@sStato) like 'ORDSPED%' 
	Set @sStile = 'orange'
Else If UPPER(@sStato) like 'FOREST%' 
	Set @sStile = 'blue, (white)'

Set @sStile = ISNULL(@sStile,'')

RETURN @sStile
END

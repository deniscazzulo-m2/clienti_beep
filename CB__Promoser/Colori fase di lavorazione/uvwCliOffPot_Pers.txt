Create View [dbo].[uvwCliOffPot_Pers] AS 
	SELECT
	    ANA.ItemDes,
	    ANA.ItemDesShort,
	    DOC.AnaType,
	    DOC.AnaCod,
		
		[dbo].[udfCliOff_TestoStato_Pers](ISNULL(DOC.Info5,'')) Info5, 
		[dbo].[udfCliOff_StileStato_Pers](ISNULL(DOC.Info5,'')) Info5Stile,

	    DOC.DocTip,
	    DOC.NumCod,
	    DOC.Numero,
	    DOC.DocNum,
	    DOC.VsRif,
	    DOC.NumExt,
	    Year(DOC.DocDta) As DocYear,
	    Month(DOC.DocDta) As DocMonth,
	    DOC.DocDta,
	    DOC.DocOra,
	    DOC.DsmCod,
	    DOC.Divisa,
	    DOC.DocTot,
	    DOC.DocImp,
	    DOC.DocIva,
	    DOC.DocOmg,
	    DOC.DocMerce,
	    DOC.DocSconti,
	    DOC.DocSpese,
	    DOC.PagCod,
	    DOC.AgeCod,
	    DOC.PrvFix,
	    DOC.PrvTot,
	    DOC.Uniq,
	    DOC.RecCreate,
	    DOC.RecUserID
	FROM dbo.[CliOff] DOC
	    Left Join dbo.[MixAna] ANA On ANA.ItemID = DOC.AnaCod And ANA.ItemType = 19
	Where (IsNull(DOC.DocKind, 0) = 19)

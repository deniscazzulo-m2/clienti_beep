CREATE FUNCTION [dbo].[udfCliOff_TestoStato_Pers]
(
	@sStato varchar(32)
)
RETURNS varchar(1024)
AS
BEGIN
Declare @sTesto varchar(64)

If UPPER(@sStato) like 'SOL%'
	Set @sTesto = REPLACE(UPPER(@sStato),'SOL ','')
Else If UPPER(@sStato) like 'ANN%'
	Set @sTesto = REPLACE(UPPER(@sStato),'ANN ','')
Else If UPPER(@sStato) like 'ACC%' 
	Set @sTesto = REPLACE(UPPER(@sStato),'ACC ','')

Set @sTesto = ISNULL(@sTesto,'')

RETURN @sTesto
END

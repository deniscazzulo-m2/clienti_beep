CREATE FUNCTION [dbo].[udfCliOff_StileStato_Pers]
(
	@sStato varchar(32)
)
RETURNS varchar(1024)
AS
BEGIN
Declare @sStile varchar(64)

If UPPER(@sStato) like 'SOL%'
	Set @sStile = 'yellow'
Else If UPPER(@sStato) like 'ANN%'
	Set @sStile = 'red'
Else If UPPER(@sStato) like 'ACC%' 
	Set @sStile = 'green, (white)'

Set @sStile = ISNULL(@sStile,'')

RETURN @sStile
END
